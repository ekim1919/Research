\begin{algorithm}[H]
\SetKwInput{KwInput}{Input}
\SetKwInput{KwOutput}{Output}
\SetKwFunction{SolveLinEq}{SolveLinearEq}

\SetAlgoLined
\DontPrintSemicolon

\KwInput{Parallelotope $P = \langle \Lambda, c_l, c_u \rangle$ in Half-Space Representation}
\KwOutput{Generator Representation of $P = \langle a, g_1 \cdots, g_n \rangle$ }
$v_0 \gets \SolveLinEq(\Lambda, c_l)$ \; \label{ln:find_anchor}
\For{$i \gets 1$ \KwTo $n$}{ \label{ln:first_loop}
  $\mu_j \gets c_l$\;
  $\mu_j[i] \gets c_u[i]$ \;
  $v_i \gets \SolveLinEq(\Lambda,\mu_i)$ \;
}
$a \gets v_0$ \;
\For{$i \gets 1$ \KwTo $n$}{
$g_i \gets v_{i} - v_0$
}
\Return{$\langle a, g_1, \cdots, g_n \rangle$}
\end{algorithm}
\caption{Half-Space to Generator Representation Conversion Algorithm.}
\label{alg:rep_conversion}
